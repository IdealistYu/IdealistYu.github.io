2020問題

2020解答

（１） \[ \begin{align} F(w)&= \begin{cases} 1-w&(0\leq w\leq1)\\ 0&(w>1) \end{cases} \\ f(t)&=\frac{2}{\pi}\int_0^{\infty}F(w)\cos\omega t\,\mathrm dw\\ &=\frac{2}{\pi}\int_0^{1}(1-w)\cos\omega t\,\mathrm dw\\ &=\frac{2}{\pi}\left[\int_0^1\cos\omega t\,\mathrm d\omega-\int_0^1\omega\cos\omega t\,\mathrm d\omega\right]\\ &=\frac{2(1-\cos t)}{\pi t^2} \end{align} \] （２） \[ \begin{align} y''-3y'+2y&=\mathrm e^{2x}\\ \\ y''-3y'+2y&=0\\ r^2-3r+2&=0\\ r_1&=1\\ r_2&=2\\ Y&=C_1\mathrm e^{x}+C_2\mathrm e^{2x}\\ \\ y^\star&=Ax\mathrm e^{2x}\\ y^\star{'}&=A\mathrm e^{2x}(1+2x)\\ y^\star{''}&=4A\mathrm e^{2x}(1+x)\\ A&=1\\ y^\star&=x\mathrm e^{2x}\\ \\ y&=Y+y^\star\\ &=C_1\mathrm e^{x}+C_2\mathrm e^{2x}+x\mathrm e^{2x} \end{align} \] （３） \[ \begin{align} z&={\mathrm e}^{i \theta}とおき,\cos\theta=\frac{z+z^{-1}}{2},\mathrm d\theta=\frac{\mathrm dz}{iz}を与式に代入すると\\ \\ I&=\int_0^{2\pi}\frac{1}{(3+\cos\theta)^2}\mathrm d\theta\\ &=\oint_{|z|=1}\frac{1}{\left(3+\frac{z+z^{-1}}{2}\right)^2}\frac{\mathrm dz}{iz}\\ &=\frac{4}{i}\oint_{|z|=1}\frac{z}{(z^2+6z+1)^2}\mathrm dz\\ &=\frac{4}{i}\oint_{|z|=1}\frac{z}{(z-z_1)^2(z-z_2)^2}\mathrm dz\\ \\ &ただし,z_1=-3+2\sqrt 2,z_2=-3-2\sqrt 2である.\\ &|z_2|>1,z_1は|z|<1内での2位の極であるから\\ \\ \mathrm {Res}(z_1)&=\left.\frac{1}{1!}\frac{\mathrm d}{\mathrm dz}\left[\frac{4}{i}\frac{z}{(z-z_2)^2}\right]\right |_{z=z_1}\\ &=\frac{3\sqrt 2}{32i}\\ したがって\\ I&=\int_0^{2\pi}\frac{1}{(3+\cos\theta)^2}\mathrm d\theta=2\pi i\,{\mathrm {Res}(z_1)}=\frac{3\sqrt 2\pi}{16} \end{align} \]

&=-\frac{1}{4ai}\oint_{|z|=1}\frac{(z^2+1)^2}{z^2\left(z-\frac{1}{a}\right)\left(z-a\right)}\frac{\mathrm dz}{iz}\\ \\ (b)\\ &ただし,z_1=0,z_2=\frac{1}{a},z_3=aである.\\ I&=\int_0^{2\pi}\frac{\cos^2\theta}{1-2a\cos\theta+a^2}\mathrm d\theta=2\pi i\,\{\mathrm {Res}(z_1)+\mathrm {Res}(z_3)\}=\frac{(1+a^2)\pi}{1-a^2} \end{align} \]

2018解答

（１） \[ \begin{align} (a)\\ F(\omega)&=\int_{-\infty}^{\infty}f(t)\mathrm e^{-i\omega t}\,\mathrm dt\\ &=\int_{-\infty}^{\infty}\mathrm e^{-a|t|}\mathrm e^{-i\omega t}\,\mathrm dt\\ &=2\int_0^{\infty}\mathrm e^{-at}\cos\omega t\,\mathrm dt\\ &=2\left.\left[\mathrm e^{-at}\frac{\omega\sin \omega t-a\cos\omega t}{a^2+\omega^2}\right]\right|_0^{\infty}\\ &=\frac{2a}{a^2+\omega^2}\\ \\ (b)\\ \mathscr F[F(t)]&=2\pi f(-\omega)=2\pi\mathrm e^{-a|\omega|}\quad(a>0)\\ \\ (c)\\ \mathscr F[h(t)]&=\int_{-\infty}^{\infty}\left(\int_{-\infty}^{\infty}f(u)g(t-u)\,\mathrm du\right)\mathrm e^{-i\omega t}\,\mathrm dt\\ &=\int_{-\infty}^{\infty}f(u)\left(\int_{-\infty}^{\infty}g(t-u)\mathrm e^{-i\omega t}\,\mathrm dt\right)\,\mathrm du\\ &=\int_{-\infty}^{\infty}f(u)F(\omega)\mathrm e^{-i\omega u}\,\mathrm du\\ &=F(\omega)G(\omega)\\ \\ (d)\\ f_1(t)&=\mathrm e^{-2|t|}とおくと\\ F_1(\omega)&=\frac{4}{\omega^2+4}\\ g(t)&=\frac{1}{t^2+1}とおくと\\ G(\omega)&=\int_{-\infty}^{\infty}\frac{1}{t^2+1}\mathrm e^{-i\omega t}\,\mathrm dt=\frac{\pi}{\mathrm e^{\omega}}\\ \mathscr F\left[\int_{-\infty}^{\infty}\frac{2f(u)}{(t-u)^2+1}\,\mathrm du\right]&=\mathscr F[F_1(t)]=2\pi f_1(-\omega)\\ 2F(\omega)G(\omega)&=2\pi\mathrm e^{-2\omega}\\ 2F(\omega)\frac{\pi}{\mathrm e^\omega}&=2\pi\mathrm e^{-2\omega}\\ F(\omega)&=\mathrm e^{-\omega}=\frac{1}{\pi}G(\omega)\\ f(t)&=\frac{1}{\pi}g(t)=\frac{1}{\pi(t^2+1)} \end{align} \] （２） \[ \begin{align} y''+\frac{x-1}{x}y'-\frac{1}{x}y&=x\mathrm e^{-x}\\ \\ y''+\frac{x-1}{x}y'-\frac{1}{x}y&=0\\ y_1&=\mathrm e^{-x}\\ \\ f(x)&=x\mathrm e^{-x}\\ p(x)&=1-\frac{1}{x}\\ q(x)&=-\frac{1}{x}\\ \\ y_2&=y_1\int y_1^{-2}\mathrm e^{-\int p(x)\,\mathrm dx}\,\mathrm dx\\ &=\mathrm e^{-x}\int \mathrm e^{2x}\mathrm e^{\int (\frac{1}{x}-1)\,\mathrm dx}\,\mathrm dx\\ &=\mathrm e^{-x}\int x\mathrm e^{x}\,\mathrm dx\\ &=x-1\\ \\ y_3&=y_1\int\left[y_1^{-2}\mathrm e^{-\int p(x)\,\mathrm dx}\int y_1f(x)\mathrm e^{\int p(x)\,\mathrm dx}\,\mathrm dx\right]\,\mathrm dx\\ &=\mathrm e^{-x}\int\left[x\mathrm e^{x}\int \mathrm e^{-x}\,\mathrm dx\right]\,\mathrm dx\\ &=-\mathrm e^{-x}\int x\,\mathrm dx\\ &=-\frac{x^2 \mathrm e^{-x}}{2}\\ \\ y&=C_1y_1+C_2y_2+y_3\\ &=C_1\mathrm e^{-x}+C_2(x-1)-\frac{x^2 \mathrm e^{-x}}{2} \end{align} \] （３） \[ \begin{align} f(x)&=\frac{x}{1+x^4}とおくと\\ I&=\int_{-\infty}^{\infty}\frac{x\sin x}{1+x^4}\,\mathrm dx\\ &=\int_{-\infty}^{\infty}f(x)\sin x\,\mathrm dx\\ &=\mathrm{Im}\int_{-\infty}^{\infty}f(x)\mathrm e^{ix}\,\mathrm dx\\ ただし,z_1&=\mathrm e^{\frac{i\pi}{4}}=\frac{1+i}{\sqrt 2},z_2=\mathrm e^{\frac{i3\pi}{4}}=\frac{-1+i}{\sqrt 2}は\,\mathrm{Im}z>0\,内の1位の極であるから\\ \\ \oint_Cf(z)\mathrm e^{iz}\,\mathrm dz&=\int_{-R}^Rf(x)\mathrm e^{ix}\,\mathrm dx+\int_{C_R}f(z)\mathrm e^{iz}\,\mathrm dz\\ &=2\pi i\{\mathrm{Res}(z_1)+\mathrm{Res}(z_2)\}\\ \\ \mathrm{Res}(z_1)&=\mathrm{Res}\left\{\frac{z\mathrm e^{iz}}{1+x^4}\right\}\\ &=\left.\frac{z\mathrm e^{iz}}{4z^3}\right|_{z=z_1}\\ &=\frac{\mathrm e^{(-1+i)/\sqrt 2}}{4i}\\ \mathrm{Res}(z_2)&=\mathrm{Res}\left\{\frac{z\mathrm e^{iz}}{1+x^4}\right\}\\ &=\left.\frac{z\mathrm e^{iz}}{4z^3}\right|_{z=z_2}\\ &=\frac{\mathrm e^{(-1-i)/\sqrt 2}}{-4i}\\ \\ &\int_{-R}^Rf(x)\mathrm e^{ix}\,\mathrm dx+\int_{C_R}f(z)\mathrm e^{iz}\,\mathrm dz=\pi i\mathrm e^{-1/{\sqrt 2}}\sin\frac{1}{\sqrt 2}\\ \because& \lim _{R\to \infty}\int_{C_R}f(z)\mathrm e^{iz}\,\mathrm dz=0\\ \therefore& \int_{-\infty}^\infty f(x)\mathrm e^{ix}\,\mathrm dx=\pi i\mathrm e^{-1/{\sqrt 2}}\sin\frac{1}{\sqrt 2}\\ \therefore& I=\mathrm{Im}(\pi i\mathrm e^{-1/{\sqrt 2}}\sin\frac{1}{\sqrt 2})=\pi \mathrm e^{-1/{\sqrt 2}}\sin\frac{1}{\sqrt 2} \end{align} \]

2016解答

（１） \[ \begin{align} (a)\\ F_1(\omega)&=\int_{-\infty}^{\infty}f_1(t)\,\mathrm e^{-i\omega t}\,\mathrm dt\\ &=\int^{\frac{\pi}{2}}_{-\frac{\pi}{2}}\cos t\,\mathrm e^{-i\omega t}\,\mathrm dt\\ &=\int^{\frac{\pi}{2}}_{-\frac{\pi}{2}}\frac{\mathrm e^{it}+\mathrm e^{-it}}{2}\,\mathrm e^{-i\omega t}\,\mathrm dt\\ &=\frac{1}{2}\int^{\frac{\pi}{2}}_{-\frac{\pi}{2}}\left[\mathrm e^{(1-\omega)it}+\mathrm e^{-(1+\omega)it}\right]\,\mathrm dt\\ &=\frac{1}{2}\int^{\frac{\pi}{2}}_{-\frac{\pi}{2}}\mathrm e^{(1-\omega)it}\,\mathrm dt+\frac{1}{2}\int^{\frac{\pi}{2}}_{-\frac{\pi}{2}}\mathrm e^{-(1+\omega)it}\,\mathrm dt\\ &=\frac{1}{2}\int^{\frac{\pi}{2}}_{-\frac{\pi}{2}}\cos((1-\omega)t)\,\mathrm dt+\frac{1}{2}\int^{\frac{\pi}{2}}_{-\frac{\pi}{2}}\cos((1+\omega)t)\,\mathrm dt\\ &=\int^{\frac{\pi}{2}}_{0}\cos((1-\omega)t)\,\mathrm dt+\int^{\frac{\pi}{2}}_{0}\cos((1+\omega)t)\,\mathrm dt\\ &=\left.\left[\frac{\sin((1-\omega)t)}{1-\omega}\right]\right|_0^{\frac{\pi}{2}}+\left.\left[\frac{\sin((1+\omega)t)}{1+\omega}\right]\right|_0^{\frac{\pi}{2}}\\ &=\frac{\cos\frac{\pi\omega}{2}}{1-\omega}+\frac{\cos\frac{\pi\omega}{2}}{1+\omega}\\ &=\frac{2\cos\frac{\pi\omega}{2}}{1-\omega^2}\\ \end{align} \]

\[ \begin{align} (b)\\ f_1(t)&=\frac{1}{2\pi}\int_{-\infty}^{\infty}F_1(\omega)\mathrm e^{i\omega t}\,\mathrm d\omega\\ &=\frac{1}{2\pi}\int_{-\infty}^{\infty}\frac{2\cos\frac{\pi\omega}{2}}{1-\omega^2}\mathrm e^{i\omega t}\,\mathrm d\omega\\ &=\int_{-\infty}^{\infty}\frac{1}{\pi(1-\omega^2)}\left(\frac{\mathrm e^{\frac{\pi i\omega}{2}}+\mathrm e^{-\frac{\pi i\omega}{2}}}{2}\right)\mathrm e^{i\omega t}\,\mathrm d\omega\\ &=\frac{1}{2}\int_{-\infty}^{\infty}\frac{1}{\pi(1-\omega^2)}\left\{\mathrm e^{i\omega (t+\frac{\pi}{2})}+\mathrm e^{i\omega (t-\frac{\pi}{2})}\right\}\,\mathrm d\omega\\ &=\int_{0}^{\infty}\frac{1}{\pi(1-\omega^2)}\left\{\mathrm e^{i\omega (t+\frac{\pi}{2})}+\mathrm e^{i\omega (t-\frac{\pi}{2})}\right\}\,\mathrm d\omega\\ &=\int_{0}^{\infty}\frac{1}{\pi(1-\omega^2)}\left\{\cos(t+\frac{\pi}{2})\omega+\cos(t-\frac{\pi}{2})\omega\right\}\,\mathrm d\omega\\ &=f_2(t) \end{align} \]

（２） \[ \begin{align} z&=\frac{1}{y}\\ \frac{\mathrm dy}{\mathrm dx}&=\frac{\mathrm dy}{\mathrm dz}\frac{\mathrm dz}{\mathrm dx}=-\frac{1}{z^2}\frac{\mathrm dz}{\mathrm dx}\\ \\ \frac{\mathrm dy}{\mathrm dx}-\frac{y}{x}-xy^2&=0\\ \frac{\mathrm dy}{\mathrm dx}-\frac{y}{x}&=xy^2\\ -\frac{1}{z^2}\frac{\mathrm dz}{\mathrm dx}-\frac{1}{z}\frac{1}{x}&=\frac{1}{z^2}x\\ \frac{\mathrm dz}{\mathrm dx}+\frac{1}{x}z&=-x\\ z'+\frac{1}{x}z&=-x\\ \\ P(x)&=\frac{1}{x}\\ Q(x)&=-x\\ z&=\mathrm e^{-\int P(x)\,\mathrm dx}(\int Q(x)\mathrm e^{\int P(x)\,\mathrm dx}\,\mathrm dx+c)\\ &=\mathrm e^{-\int \frac{1}{x}\,\mathrm dx}(\int -x\mathrm e^{\int \frac{1}{x}\,\mathrm dx}\,\mathrm dx+c)\\ &=\frac{1}{x}(\int -x\mathrm e^{\int \frac{1}{x}\,\mathrm dx}\,\mathrm dx+c)\\ &=\frac{1}{x}(\int -x^2\,\mathrm dx+c)\\ &=\frac{1}{x}( \frac{-x^3}{3}+c)\\ &=\frac{C-x^3}{3x}\\ \\ y&=\frac{1}{z}=\frac{3x}{C-x^3} \end{align} \] （３） \[ \begin{align} (a)\\ z&={\mathrm e}^{i\theta}とおき,\sin\theta=\frac{z-z^{-1}}{2i},\mathrm d\theta=\frac{\mathrm dz}{iz}を与式に代入すると\\ \\ I&=\int_0^{2\pi}\frac{1}{a+b\sin\theta}\mathrm d\theta\\ &=\oint_{|z|=1}\frac{1}{a+b\left(\frac{z-z^{-1}}{2i}\right)}\frac{\mathrm dz}{iz}\\ &=\frac{2}{b}\oint_{|z|=1}\frac{1}{z^2+\frac{2ai}{b}z-1}\mathrm dz\\ \\ (b)\\ I&=\frac{2}{b}\oint_{|z|=1}\frac{1}{z^2+\frac{2ai}{b}z-1}\mathrm dz\\ &=\frac{2}{b}\oint_{|z|=1}\frac{1}{(z-\alpha)(z-\beta)}\mathrm dz\\ \\ &ただし,z_1=\alpha=\frac{-a+\sqrt{a^2-b^2}}{b}i,z_2=\beta=\frac{-a-\sqrt{a^2-b^2}}{b}iである.\\ &|z_2|>1,z_1は|z|<1内での1位の極であるから\\ \\ \mathrm {Res}(z_1)&=\left.\frac{2}{b}\frac{1}{z-\beta}\right |_{z=z_1}\\ &=\frac{1}{\sqrt{a^2-b^2}i}\\ したがって\\ I&=\int_0^{2\pi}\frac{1}{a+b\sin\theta}\mathrm d\theta=2\pi i\,\mathrm {Res}(z_1)=\frac{2\pi}{\sqrt{a^2-b^2}} \end{align} \]
